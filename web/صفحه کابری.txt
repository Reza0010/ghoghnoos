        // ماژول مدیریت اعلان‌ها
        const NotificationModule = (function() {
            const notification = document.getElementById('notification');
            const notificationText = document.getElementById('notificationText');
            let timeoutId;
            
            /**
             * نمایش اعلان به کاربر
             * @param {string} message - پیام اعلان
             * @param {string} type - نوع اعلان (success, error, info)
             * @param {number} duration - مدت زمان نمایش به میلی‌ثانیه
             */
            function show(message, type = 'success', duration = 3000) {
                // پاک کردن تایمر قبلی
                if (timeoutId) {
                    clearTimeout(timeoutId);
                }
                
                notificationText.textContent = message;
                notification.className = 'notification show ' + type;
                
                // به‌روزرسانی آیکون
                const icon = notification.querySelector('i');
                icon.className = type === 'success' ? 'fas fa-check-circle' : 
                                 type === 'error' ? 'fas fa-exclamation-circle' : 
                                 'fas fa-info-circle';
                
                // مخفی شدن خودکار
                timeoutId = setTimeout(() => {
                    hide();
                }, duration);
            }
            
            function hide() {
                notification.classList.remove('show');
            }
            
            return {
                show: show,
                hide: hide
            };
        })();

                    const filename = `dashboard_export_${new Date().toISOString().slice(0,10)}`;

                        case 'pdf':

                            break;

            function downloadCSV(data, filename) {
                const csv = [
                    Object.keys(data[0]).join(','),
                    ...data.map(row => Object.values(row).join(','))
                ].join('\n');
                
                downloadFile(csv, `${filename}.csv`, 'text/csv');
                NotificationModule.show('فایل CSV با موفقیت دانلود شد', 'success');
            }
            
            function downloadJSON(data, filename) {
                const json = JSON.stringify(data, null, 2);
                downloadFile(json, `${filename}.json`, 'application/json');
                NotificationModule.show('فایل JSON با موفقیت دانلود شد', 'success');
            }
            
            function downloadFile(content, filename, mimeType) {
                const blob = new Blob([content], { type: mimeType });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = filename;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                window.URL.revokeObjectURL(url);
            }
            
            return {
                init: init
            };
        })();

            /**
             * اتصال رویدادهای صفحه
             * @param {string} page - نام صفحه
             */
            function bindPageEvents(page) {
                // رویدادهای کارت‌های میز کار
                if (page === 'workspace') {
                    document.querySelectorAll('.workspace-card').forEach(card => {
                        card.addEventListener('click', function() {
                            const cardType = this.getAttribute('data-card');
                            NotificationModule.show(`در حال باز کردن ${this.querySelector('.card-title').textContent}...`, 'info');
                        });
                        
                        card.addEventListener('keydown', function(e) {
                            if (e.key === 'Enter' || e.key === ' ') {
                                e.preventDefault();
                                this.click();
                            }
                        });
                    });
                    
                    // اتصال مجدد دکمه‌های صادرات
                    ExportModule.init();
                }
                
                // رویدادهای فرم پروفایل
                if (page === 'profile') {
                    const saveButton = document.getElementById('saveProfile');
                    if (saveButton) {
                        saveButton.addEventListener('click', saveProfile);
                    }
                }
                
                // رویدادهای فرم آدرس
                if (page === 'addresses') {
                    const saveButton = document.getElementById('saveAddress');
                    if (saveButton) {
                        saveButton.addEventListener('click', saveAddress);
                    }
                    
                    // اتصال رویداد تغییر استان
                    const provinceSelect = document.getElementById('province');
                    const citySelect = document.getElementById('city');
                    
                    provinceSelect.addEventListener('change', function() {
                        const selectedProvince = this.value;
                        
                        // خالی کردن و غیرفعال کردن شهر
                        citySelect.innerHTML = '<option value="">در حال بارگذاری...</option>';
                        citySelect.disabled = true;
                        
                        if (selectedProvince && provincesAndCities[selectedProvince]) {
                            // شبیه‌سازی بارگذاری شهرها
                            setTimeout(() => {
                                citySelect.innerHTML = '<option value="">شهر خود را انتخاب کنید</option>';
                                provincesAndCities[selectedProvince].forEach(city => {
                                    const option = document.createElement('option');
                                    option.value = city;
                                    option.textContent = city;
                                    citySelect.appendChild(option);
                                });
                                citySelect.disabled = false;
                            }, 300);
                        } else {
                            citySelect.innerHTML = '<option value="">ابتدا استان را انتخاب کنید</option>';
                            citySelect.disabled = true;
                        }
                    });
                }
            }
            
            /**
             * ذخیره اطلاعات پروفایل
             */
            function saveProfile() {
                const fields = {
                    firstName: {
                        value: document.getElementById('firstName').value.trim(),
                        error: 'لطفاً نام خود را وارد کنید',
                        validate: v => v.length > 0
                    },
                    lastName: {
                        value: document.getElementById('lastName').value.trim(),
                        error: 'لطفاً نام خانوادگی خود را وارد کنید',
                        validate: v => v.length > 0
                    },
                    mobile: {
                        value: document.getElementById('mobile').value.trim(),
                        error: 'لطفاً یک شماره همراه معتبر وارد کنید',
                        validate: v => /^09[0-9]{9}$/.test(v.replace(/\s/g, ''))
                    },
                    email: {
                        value: document.getElementById('email').value.trim(),
                        error: 'لطفاً یک پست الکترونیکی معتبر وارد کنید',
                        validate: v => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v)
                    }
                };
                
                let isValid = true;
                
                // اعتبارسنجی فیلدها
                for (const [fieldId, field] of Object.entries(fields)) {
                    const input = document.getElementById(fieldId);
                    const errorElement = document.getElementById(`${fieldId}Error`);
                    
                    if (!field.validate(field.value)) {
                        input.classList.add('error');
                        errorElement.textContent = field.error;
                        errorElement.classList.add('show');
                        isValid = false;
                    } else {
                        input.classList.remove('error');
                        errorElement.classList.remove('show');
                    }
                }
                
                if (!isValid) {
                    NotificationModule.show('لطفاً تمام فیلدهای ضروری را صحیح وارد کنید', 'error');
                    return;
                }
                
                // شبیه‌سازی ذخیره اطلاعات
                LoadingModule.show();
                setTimeout(() => {
                    LoadingModule.hide();
                    NotificationModule.show('اطلاعات با موفقیت ذخیره شد', 'success');
                    
                    // به‌روزرسانی نام کاربر در نوار بالا
                    document.querySelector('.user-name').textContent = fields.firstName.value + ' ' + fields.lastName.value;
                }, 1000);
            }
            
            /**
             * ذخیره آدرس
             */
            function saveAddress() {
                const fields = {
                    province: {
                        value: document.getElementById('province').value,
                        error: 'لطفاً استان خود را انتخاب کنید',
                        validate: v => v.length > 0
                    },
                    city: {
                        value: document.getElementById('city').value,
                        error: 'لطفاً شهر خود را انتخاب کنید',
                        validate: v => v.length > 0
                    },
                    address: {
                        value: document.getElementById('address').value.trim(),
                        error: 'لطفاً آدرس دقیق خود را وارد کنید',
                        validate: v => v.length > 0
                    },
                    postalCode: {
                        value: document.getElementById('postalCode').value.trim(),
                        error: 'لطفاً یک کد پستی معتبر (۱۰ رقم) وارد کنید',
                        validate: v => /^\d{10}$/.test(v.replace(/\s/g, '').replace(/-/g, ''))
                    }
                };
                
                let isValid = true;
                
                // اعتبارسنجی فیلدها
                for (const [fieldId, field] of Object.entries(fields)) {
                    const input = document.getElementById(fieldId);
                    const errorElement = document.getElementById(`${fieldId}Error`);
                    
                    if (!field.validate(field.value)) {
                        input.classList.add('error');
                        errorElement.textContent = field.error;
                        errorElement.classList.add('show');
                        isValid = false;
                    } else {
                        input.classList.remove('error');
                        errorElement.classList.remove('show');
                    }
                }
                
                if (!isValid) {
                    NotificationModule.show('لطفاً تمام فیلدهای ضروری را صحیح وارد کنید', 'error');
                    return;
                }
                
                // شبیه‌سازی ذخیره آدرس
                LoadingModule.show();
                setTimeout(() => {
                    LoadingModule.hide();
                    NotificationModule.show('آدرس با موفقیت ذخیره شد', 'success');
                    
                    // خالی کردن فرم پس از ذخیره
                    document.getElementById('province').value = '';
                    document.getElementById('city').innerHTML = '<option value="">ابتدا استان را انتخاب کنید</option>';
                    document.getElementById('city').disabled = true;
                    document.getElementById('address').value = '';
                    document.getElementById('postalCode').value = '';
                }, 1000);
            }
            
            /**
             * مدیریت رویدادهای منوی کشویی کاربر
             */
            function bindUserDropdownEvents() {
                // منوی کشویی کاربر
                userInfo.addEventListener('click', function(e) {
                    e.stopPropagation();
                    this.classList.toggle('active');
                    dropdownMenu.classList.toggle('show');
                    
                    // به‌روزرسانی ویژگی‌های ARIA
                    const expanded = this.classList.contains('active');
                    this.setAttribute('aria-expanded', expanded);
                });
                
                // پشتیبانی از کیبورد
                userInfo.addEventListener('keydown', function(e) {
                    if (e.key === 'Enter' || e.key === ' ') {
                        e.preventDefault();
                        this.click();
                    } else if (e.key === 'Escape' && this.classList.contains('active')) {
                        this.classList.remove('active');
                        dropdownMenu.classList.remove('show');
                        this.setAttribute('aria-expanded', 'false');
                    }
                });
                
                // کلیک در جای دیگر برای بستن منوی کشویی
                document.addEventListener('click', function() {
                    userInfo.classList.remove('active');
                    dropdownMenu.classList.remove('show');
                    userInfo.setAttribute('aria-expanded', 'false');
                });
                
                // جلوگیری از انتشار رویداد کلیک در داخل منوی کشویی
                dropdownMenu.addEventListener('click', function(e) {
                    e.stopPropagation();
                });
                
                // رویداد کلیک آیتم‌های منوی کشویی
                document.querySelectorAll('.dropdown-item').forEach(item => {
                    item.addEventListener('click', function(e) {
                        e.preventDefault();
                        const action = this.getAttribute('data-action');
                        
                        userInfo.classList.remove('active');
                        dropdownMenu.classList.remove('show');
                        userInfo.setAttribute('aria-expanded', 'false');
                        
                        switch(action) {
                            case 'profile':
                                loadPageContent('profile');
                                break;
                            case 'password':
                                NotificationModule.show('در حال باز کردن صفحه تغییر رمز عبور...', 'info');
                                break;
                            case 'settings':
                                NotificationModule.show('در حال باز کردن صفحه تنظیمات...', 'info');
                                break;
                            case 'logout':
                                if (confirm('آیا از خروج اطمینان دارید؟')) {
                                    NotificationModule.show('شما با موفقیت خارج شدید', 'success');
                                    setTimeout(() => {
                                        window.location.href = '#';
                                    }, 2000);
                                }
                                break;
                        }
                    });
                    
                    // پشتیبانی از کیبورد
                    item.addEventListener('keydown', function(e) {
                        if (e.key === 'Enter' || e.key === ' ') {
                            e.preventDefault();
                            this.click();
                        }
                    });
                });
            }
            
            /**
             * مدیریت رویدادهای نوار کناری
             */
            function bindSidebarEvents() {
                document.querySelectorAll('.left-sidebar-item').forEach(item => {
                    item.addEventListener('click', function(e) {
                        e.preventDefault();
                        const page = this.getAttribute('data-page');
                        
                        // به‌روزرسانی وضعیت فعال
                        document.querySelectorAll('.left-sidebar-item').forEach(i => {
                            i.classList.remove('active');
                        });
                        this.classList.add('active');
                        
                        // بارگذاری محتوای صفحه
                        if (page === 'site') {
                            NotificationModule.show('در حال بازگشت به سایت...', 'info');
                            setTimeout(() => {
                                window.location.href = '#';
                            }, 1500);
                        } else {
                            loadPageContent(page);
                        }
                    });
                    
                    // پشتیبانی از کیبورد
                    item.addEventListener('keydown', function(e) {
                        if (e.key === 'Enter' || e.key === ' ') {
                            e.preventDefault();
                            this.click();
                        }
                    });
                });
            }
            
            /**
             * مقداردهی اولیه برنامه
             */
            function init() {
                // مقداردهی اولیه ماژول‌ها
                ThemeModule.init();
                SearchModule.init();
                ExportModule.init();
                
                // اتصال رویدادها
                bindUserDropdownEvents();
                bindSidebarEvents();
                
                // مدیریت خطاهای جهانی
                window.addEventListener('error', function(e) {
                    console.error('خطای جهانی:', e.error);
                    NotificationModule.show('خطایی رخ داد. لطفاً بعداً دوباره تلاش کنید.', 'error');
                });
            }
            
            // API عمومی
            return {
                init: init,
                loadPageContent: loadPageContent
            };
        })();
        
        // مقداردهی اولیه برنامه
        document.addEventListener('DOMContentLoaded', function() {
            DashboardApp.init();
        });
    </script>
