    <script>
        document.addEventListener('DOMContentLoaded', function() {
            
            const masonryGrid = document.getElementById('masonry-grid');
            const emptyWishlist = document.getElementById('empty-wishlist');
            const pageTitle = document.getElementById('page-title');
            const pageSubtitle = document.getElementById('page-subtitle');
            const wishlistList = document.getElementById('wishlist-list');

            // --- عناصر فرم جدید ---
            const createListBtn = document.getElementById('create-list-btn');
            const createFirstListBtn = document.getElementById('create-first-list-btn');
            const newListForm = document.getElementById('new-list-form');
            const newListNameInput = document.getElementById('new-list-name-input');
            const confirmNewListBtn = document.getElementById('confirm-new-list-btn');
            const cancelNewListBtn = document.getElementById('cancel-new-list-btn');

            // --- داده‌های نمونه برای علاقه‌مندی‌ها ---
            const wishlists = {
                "عمومی": [
                    { id: 'w1', name: 'مانتو کتی کلاسیک', price: 450000, image: 'https://picsum.photos/seed/wish1/400/600.jpg' },
                    { id: 'w2', name: 'شال حریر ابریشمی', price: 250000, image: 'https://picsum.photos/seed/wish2/400/300.jpg' },
                    { id: 'w3', name: 'ست بلوز و شلوارک', price: 490000, image: 'https://picsum.photos/seed/wish3/400/500.jpg' },
                    { id: 'w4', name: 'کاپشن بهاری', price: 550000, image: 'https://picsum.photos/seed/wish4/400/450.jpg' },
                    { id: 'w5', name: 'مانتو مجلسی بلند', price: 750000, image: 'https://picsum.photos/seed/wish5/400/550.jpg' },
                ],
                "ایده‌های هدیه": [
                    { id: 'w6', name: 'جعبه آرایشی', price: 350000, image: 'https://picsum.photos/seed/wish6/400/400.jpg' },
                    { id: 'w7', name: 'عطر زنانه', price: 850000, image: 'https://picsum.photos/seed/wish7/400/350.jpg' },
                ]
            };
            let currentListName = Object.keys(wishlists)[0] || null;

            // --- توابع کمکی ---
            function showToast(message) {
                const toastEl = document.createElement('div');
                toastEl.className = 'toast';
                toastEl.innerHTML = `<i class="fas fa-check-circle"></i> ${message}`;
                document.body.appendChild(toastEl);
                setTimeout(() => toastEl.classList.add('show'), 10);
                setTimeout(() => {
                    toastEl.classList.remove('show');
                    setTimeout(() => document.body.removeChild(toastEl), 500);
                }, 3000);
            }

            function renderWishlistList() {
                wishlistList.innerHTML = '';
                const listNames = Object.keys(wishlists);
                
                if (listNames.length === 0) {
                    wishlistList.innerHTML = '<li style="padding: 20px; text-align: center; color: #aaa;">لیستی یافت نشد</li>';
                    return;
                }

                listNames.forEach(listName => {
                    const li = document.createElement('li');
                    li.dataset.listName = listName;
                    if (listName === currentListName) {
                        li.classList.add('active');
                    }
                    li.innerHTML = `
                        <span>${listName}</span>
                        <div class="list-actions">
                            <button class="edit" title="تغییر نام"><i class="fas fa-edit"></i></button>
                            <button class="delete" title="حذف لیست"><i class="fas fa-trash"></i></button>
                        </div>
                    `;
                    li.addEventListener('click', (e) => {
                        if (!e.target.closest('.list-actions')) {
                            switchToList(listName);
                        }
                    });
                    wishlistList.appendChild(li);
                });
            }

            function renderMasonryGrid() {
                masonryGrid.innerHTML = '';
                const items = currentListName ? wishlists[currentListName] : [];

                if (items.length === 0) {
                    emptyWishlist.style.display = 'block';
                    masonryGrid.style.display = 'none';
                } else {
                    emptyWishlist.style.display = 'none';
                    masonryGrid.style.display = 'block';

                    items.forEach((item, index) => {
                        const card = document.createElement('div');
                        card.className = 'wishlist-item-card';
                        if (index === 0) card.classList.add('hero'); // اولین محصول را Hero می‌کند
                        card.dataset.id = item.id;
                        card.innerHTML = `
                            <div class="image-container">
                                <img src="${item.image}" alt="${item.name}">
                                <div class="item-overlay">
                                    <button class="overlay-btn add-to-cart-btn">افزودن به سبد</button>
                                    <button class="overlay-btn delete" title="حذف از لیست">
                                        <i class="fas fa-times"></i>
                                    </button>
                                </div>
                            </div>
                            <div class="card-body">
                                <h4>${item.name}</h4>
                                <p class="price">${Number(item.price).toLocaleString('fa-IR')} تومان</p>
                            </div>
                        `;
                        masonryGrid.appendChild(card);
                    });

                    attachItemEvents();
                }
            }

            function attachItemEvents() {
                document.querySelectorAll('.add-to-cart-btn').forEach(button => {
                    button.addEventListener('click', (e) => {
                        e.stopPropagation();
                        const itemName = e.target.closest('.wishlist-item-card').querySelector('h4').textContent;
                        showToast(`${itemName} به سبد خرید اضافه شد.`);
                    });
                });

                document.querySelectorAll('.overlay-btn.delete').forEach(button => {
                    button.addEventListener('click', (e) => {
                        e.stopPropagation();
                        const card = e.target.closest('.wishlist-item-card');
                        removeItem(card);
                    });
                });
            }

            function removeItem(card) {
                card.classList.add('hiding');
                setTimeout(() => {
                    const itemId = card.dataset.id;
                    const index = wishlists[currentListName].findIndex(item => item.id === itemId);
                    if (index > -1) {
                        wishlists[currentListName].splice(index, 1);
                    }
                    renderMasonryGrid();
                }, 400);
            }

            function switchToList(listName) {
                currentListName = listName;
                pageTitle.textContent = listName;
                pageSubtitle.textContent = `کالکشن شخصی و الهام‌بخش "${listName}".`;
                renderWishlistList();
                renderMasonryGrid();
            }
            
            // --- مدیریت فرم ایجاد لیست جدید (Inline) ---
            function showNewListForm() {
                newListForm.style.display = 'block';
                wishlistList.style.display = 'none';
                createListBtn.classList.add('active');
                createListBtn.innerHTML = '<i class="fas fa-times-circle"></i>';
                newListNameInput.value = '';
                newListNameInput.focus();
            }

            function hideNewListForm() {
                newListForm.style.display = 'none';
                wishlistList.style.display = 'block';
                createListBtn.classList.remove('active');
                createListBtn.innerHTML = '<i class="fas fa-plus-circle"></i>';
            }

            createListBtn.addEventListener('click', () => {
                if (createListBtn.classList.contains('active')) {
                    hideNewListForm();
                } else {
                    showNewListForm();
                }
            });
            createFirstListBtn.addEventListener('click', showNewListForm);
            
            cancelNewListBtn.addEventListener('click', hideNewListForm);
            
            confirmNewListBtn.addEventListener('click', () => {
                const listName = newListNameInput.value.trim();
                if (listName && !wishlists[listName]) {
                    wishlists[listName] = [];
                    hideNewListForm();
                    renderWishlistList();
                    switchToList(listName);
                    showToast(`لیست "${listName}" با موفقیت ایجاد شد.`);
                } else {
                    alert('نام لیست معتبر نیست یا از قبل وجود دارد.');
                }
            });

            // رویدادهای کلیک برای دکمه‌های ویرایش و حذف لیست
            document.addEventListener('click', (e) => {
                if (e.target.closest('.list-actions .delete')) {
                    const li = e.target.closest('li');
                    const listName = li.dataset.listName;
                    if (confirm(`آیا از حذف لیست "${listName}" مطمئن هستید؟`)) {
                        delete wishlists[listName];
                        if (currentListName === listName) {
                            const remainingLists = Object.keys(wishlists);
                            currentListName = remainingLists.length > 0 ? remainingLists[0] : null;
                        }
                        renderWishlistList();
                        renderMasonryGrid();
                        showToast(`لیست "${listName}" حذف شد.`);
                    }
                }
                if (e.target.closest('.list-actions .edit')) {
                    const li = e.target.closest('li');
                    const oldName = li.dataset.listName;
                    const newName = prompt('نام جدید را وارد کنید:', oldName);
                    if (newName && newName !== oldName && !wishlists[newName]) {
                        wishlists[newName] = wishlists[oldName];
                        delete wishlists[oldName];
                        if (currentListName === oldName) currentListName = newName;
                        renderWishlistList();
                        renderMasonryGrid();
                    }
                }
            });

            // --- مقداردهی اولیه ---
            renderWishlistList();
            renderMasonryGrid();
        });
    </script>
