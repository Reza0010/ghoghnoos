    <script>
        // --- داده‌های نمونه سبد خرید ---
        let cartItems = [
            { id: 1, name: 'مانتو کتی حریر', price: 850000, quantity: 1, image: 'https://via.placeholder.com/150' },
            { id: 2, name: 'شال و روسری ابریشمی', price: 450000, quantity: 2, image: 'https://via.placeholder.com/150' }
        ];

        let discount = 0;
        const SHIPPING_COST = 50000;
        let currentStep = 1;

        // --- توابع مربوط به مراحل ---
        function showStep(n) {
            const steps = document.getElementsByClassName("step");
            for (let i = 0; i < steps.length; i++) {
                steps[i].classList.remove('active');
            }
            steps[n - 1].classList.add('active');
            updateStepper(n);

            document.getElementById("prevBtn").disabled = (n === 1);
            const nextBtn = document.getElementById("nextBtn");
            const navigation = document.querySelector('.step-navigation');

            if (n === steps.length) { // مرحله پایانی
                navigation.style.display = 'none'; // مخفی کردن دکمه‌ها
            } else {
                navigation.style.display = 'flex';
                if (n === steps.length - 1) { // مرحله قبل از پایانی (پرداخت)
                    nextBtn.innerHTML = "ثبت نهایی سفارش";
                    nextBtn.classList.add('btn-submit');
                    nextBtn.classList.remove('btn-next');
                    populateFinalSummary();
                } else {
                    nextBtn.innerHTML = "بعدی";
                    nextBtn.classList.remove('btn-submit');
                    nextBtn.classList.add('btn-next');
                }
            }
        }

        function changeStep(n) {
            if (n === 1 && !validateStep(currentStep)) return;
            currentStep += n;
            showStep(currentStep);
        }
        
        function handleNext() {
            if (currentStep === 3) { // اگر روی دکمه "ثبت نهایی" کلیک شد
                submitOrder();
            } else {
                changeStep(1); // رفتن به مرحله بعد
            }
        }

        function validateStep(step) {
            if (step === 2) {
                const name = document.getElementById('name').value;
                const phone = document.getElementById('phone').value;
                const address = document.getElementById('address').value;
                if (!name || !phone || !address) {
                    alert('لطفا تمام فیلدهای اطلاعات ارسال را تکمیل کنید.');
                    return false;
                }
            }
            return true;
        }

        function updateStepper(step) {
            const steppers = document.getElementsByClassName('stepper-item');
            for (let i = 0; i < steppers.length; i++) {
                steppers[i].classList.remove('active', 'completed');
                if (i < step - 1) {
                    steppers[i].classList.add('completed');
                } else if (i === step - 1) {
                    steppers[i].classList.add('active');
                }
            }
        }
        
        function submitOrder() {
            // شبیه‌سازی ارسال اطلاعات به سرور و دریافت پاسخ موفق
            console.log('در حال ارسال سفارش به سرور...');
            const orderData = {
                items: cartItems,
                shipping: { /* ... */ },
                payment: document.querySelector('input[name="payment"]:checked').value,
                total: document.getElementById('final-total').textContent
            };
            console.log('اطلاعات نهایی سفارش:', orderData);

            // تولید شماره سفارش تصادفی
            const orderNumber = Math.floor(100000 + Math.random() * 900000);
            document.getElementById('order-number').textContent = orderNumber;

            // رفتن به مرحله پایانی
            currentStep++;
            showStep(currentStep);
        }
        
        // --- توابع مربوط به سبد خرید و هزینه‌ها (مشابه قبل) ---
        function renderCartItems() { /* ... */ }
        function changeQuantity(id, delta) { /* ... */ }
        function setQuantity(id, value) { /* ... */ }
        function removeItem(id) { /* ... */ }
        function updateTotals() { /* ... */ }
        function populateFinalSummary() { /* ... */ }
        
        // --- پیاده‌سازی توابع ---
        function renderCartItems() {
            const container = document.getElementById('cart-items-container');
            container.innerHTML = '';
            if (cartItems.length === 0) { container.innerHTML = '<p style="text-align:center;">سبد خرید شما خالی است.</p>'; updateTotals(); return; }
            cartItems.forEach(item => {
                container.innerHTML += `
                    <div class="cart-item">
                        <img src="${item.image}" alt="${item.name}">
                        <div class="item-details"><h4>${item.name}</h4><p>قیمت واحد: ${item.price.toLocaleString('fa-IR')} تومان</p></div>
                        <div class="item-quantity"><button onclick="changeQuantity(${item.id}, -1)">-</button><input type="number" id="qty-${item.id}" value="${item.quantity}" min="1" onchange="setQuantity(${item.id}, this.value)"><button onclick="changeQuantity(${item.id}, 1)">+</button></div>
                        <div class="item-total">${(item.price * item.quantity).toLocaleString('fa-IR')} تومان</div>
                        <button class="remove-item" onclick="removeItem(${item.id})">&times;</button>
                    </div>`;
            });
            updateTotals();
        }
        function changeQuantity(id, delta) { const item = cartItems.find(i => i.id === id); if (item) { item.quantity = Math.max(1, item.quantity + delta); renderCartItems(); } }
        function setQuantity(id, value) { const item = cartItems.find(i => i.id === id); if (item) { item.quantity = Math.max(1, parseInt(value) || 1); renderCartItems(); } }
        function removeItem(id) { cartItems = cartItems.filter(i => i.id !== id); renderCartItems(); }
        function updateTotals() {
            const subtotal = cartItems.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            const shipping = cartItems.length > 0 ? SHIPPING_COST : 0;
            const discountAmount = subtotal * (discount / 100);
            const finalTotal = subtotal + shipping - discountAmount;
            document.getElementById('subtotal').textContent = `${subtotal.toLocaleString('fa-IR')} تومان`;
            document.getElementById('shipping-cost').textContent = `${shipping.toLocaleString('fa-IR')} تومان`;
            document.getElementById('discount-amount').textContent = `${discountAmount.toLocaleString('fa-IR')} تومان`;
            document.getElementById('final-total').textContent = `${finalTotal.toLocaleString('fa-IR')} تومان`;
        }
        function populateFinalSummary() {
            const finalTotalText = document.getElementById('final-total').textContent;
            const itemCount = cartItems.reduce((sum, item) => sum + item.quantity, 0);
            document.getElementById('final-summary-content').innerHTML = `<p><strong>تعداد کالا:</strong> ${itemCount} عدد</p><p><strong>مبلغ نهایی:</strong> ${finalTotalText}</p>`;
        }

        // --- مدیریت رویدادها ---
        document.getElementById('apply-discount-btn').addEventListener('click', () => {
            const code = document.getElementById('discount-code').value.trim();
            if (code === 'OFF10') { discount = 10; alert('کد تخفیف با موفقیت اعمال شد!'); }
            else if (code === 'OFF20') { discount = 20; alert('کد تخفیف ویژه با موفقیت اعمال شد!'); }
            else { discount = 0; alert('کد تخفیف نامعتبر است.'); }
            updateTotals();
        });

        // --- بارگذاری اولیه صفحه ---
        document.addEventListener('DOMContentLoaded', () => {
            renderCartItems();
            showStep(currentStep);
        });
    </script>
